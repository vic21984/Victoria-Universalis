		part_of_sphere = yes
		sphere_owner = {
			is_culture_group = THIS
			is_cultural_union = yes
			war = no
		}
		NOT = {
			AND = {
				OR = {
					tag = AUS
					tag = KUK
				}
				sphere_owner = {
					tag = GER
				}
				NOT = {
					exists = HUN
				}
			}
		}
		NOT = {	
			tag = ISR
		}
		NOT = {	
			tag = CSA
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = "EVTOPTA60120"
		set_country_flag = soi_annexed
		sphere_owner = {
			country_event = { id=60130 days=0 }
		}
	}
}
