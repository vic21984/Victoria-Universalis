	title = "EVTNAME91707"
	desc = "EVTDESC91707"

	trigger = {
		owner = {
			NOT = { has_country_modifier = diplomatic_cores }
			war = no
		}
		has_province_modifier = adding_cores
	}

	option = {
		name = "EVTOPTA91707"
		state_scope = {
			any_owned = {
				limit = {
					has_province_modifier = adding_cores
				}
				remove_province_modifier = adding_cores
				add_core = THIS
				add_core = RMC
			}
		}
		any_country = {
			limit = {
				any_core = {
					is_core = RMC
					owned_by = THIS
				}
				NOT = { tag = THIS }
				NOT = { tag = MST }
				NOT = { tag = SMA }
			}
			country_event = { id = 91708 days = 0 }
		}
	}
}
